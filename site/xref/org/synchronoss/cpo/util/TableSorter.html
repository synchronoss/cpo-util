31  /**
32   * A sorter for TableModels. The sorter has a model (conforming to TableModel)
33   * and itself implements TableModel. TableSorter does not store or copy
34   * the data in the TableModel, instead it maintains an array of
35   * integers which it keeps the same size as the number of rows in its
36   * model. When the model changes it notifies the sorter that something
37   * has changed eg. "rowsAdded" so that its internal array of integers
38   * can be reallocated. As requests are made of the sorter (like
39   * getValueAt(row, col) it redirects them to its model via the mapping
40   * array. That way the TableSorter appears to hold another copy of the table
41   * with the rows in a different order. The sorting algorthm used is stable
42   * which means that it does not move around rows when its comparison
43   * function returns 0 to denote that they are equivalent.
44   *
45   * @version 1.5 12/17/97
46   * @author Philip Milne
47   */
48  public class TableSorter extends TableMap {

68    public int compareRowsByColumn(int row1, int row2, int column) {
69      Class<?> type = model.getColumnClass(column);
70      TableModel data = model;
71  
72      // Check for nulls.
73  
74      Object o1 = data.getValueAt(row1, column);
75      Object o2 = data.getValueAt(row2, column);
76  
77      // If both values are null, return 0.
78      if (o1 == null && o2 == null) {
79        return 0;
80      } else if (o1 == null) { // Define null less than everything.
81        return -1;
82      } else if (o2 == null) {
83        return 1;
84      }
85  
86      /*
87       * We copy all returned values from the getValue call in case
88       * an optimised model is reusing one object to return many
89       * values.  The Number subclasses in the JDK are immutable and
90       * so will not be used in this way but other subclasses of
91       * Number might want to do this to save space and avoid
92       * unnecessary heap allocation.
93       */
94      if (type.getSuperclass() == java.lang.Number.class) {
95        Number n1 = (Number)data.getValueAt(row1, column);
96        double d1 = n1.doubleValue();
97        Number n2 = (Number)data.getValueAt(row2, column);
98        double d2 = n2.doubleValue();
99  
100       if (d1 < d2) {
101         return -1;
102       } else if (d1 > d2) {
103         return 1;
104       } else {
105         return 0;
106       }
107     } else if (type == java.util.Date.class) {
108       Date d1 = (Date)data.getValueAt(row1, column);
109       long n1 = d1.getTime();
110       Date d2 = (Date)data.getValueAt(row2, column);
111       long n2 = d2.getTime();
112 
113       if (n1 < n2) {
114         return -1;
115       } else if (n1 > n2) {
116         return 1;
117       } else {
118         return 0;
119       }
120     } else if (type == String.class) {
121       String s1 = (String)data.getValueAt(row1, column);
122       String s2 = (String)data.getValueAt(row2, column);
123       int result = s1.compareTo(s2);
124 
125       if (result < 0) {
126         return -1;
127       } else if (result > 0) {
128         return 1;
129       } else {
130         return 0;
131       }
132     } else if (type == Boolean.class) {
133       Boolean b1 = (Boolean)data.getValueAt(row1, column);
134       Boolean b2 = (Boolean)data.getValueAt(row2, column);
135 
136       if (b1.booleanValue() == b2.booleanValue()) {
137         return 0;
138       } else if (b1) { // Define false < true
139         return 1;
140       } else {
141         return -1;
142       }
143     } else {
144       Object v1 = data.getValueAt(row1, column);
145       String s1 = v1.toString();
146       Object v2 = data.getValueAt(row2, column);
147       String s2 = v2.toString();
148       int result = s1.compareTo(s2);
149 
150       if (result < 0) {
151         return -1;
152       } else if (result > 0) {
153         return 1;
154       } else {
155         return 0;
156       }
157     }
158   }

160   public int compare(int row1, int row2) {
161     for (int level = 0; level < sortingColumns.size(); level++) {
162       Integer column = sortingColumns.elementAt(level);
163       int result = compareRowsByColumn(row1, row2, column);
164       if (result != 0) {
165         return ascending ? result : -result;
166       }
167     }
168     return 0;
169   }

171   public void reallocateIndexes() {
172     int rowCount = model.getRowCount();
173 
174     // Set up a new array of indexes with the right number of elements
175     // for the new data model.
176     indexes = new int[rowCount];
177 
178     // Initialise with the identity mapping.
179     for (int row = 0; row < rowCount; row++) {
180       indexes[row] = row;
181     }
182   }

202   public void n2sort() {
203     for (int i = 0; i < getRowCount(); i++) {
204       for (int j = i + 1; j < getRowCount(); j++) {
205         if (compare(indexes[i], indexes[j]) == -1) {
206           swap(i, j);
207         }
208       }
209     }
210   }

212   /**
213    * This is a home-grown implementation which we have not had time
214    * to research - it may perform poorly in some circumstances. It
215    * requires twice the space of an in-place algorithm and makes
216    * NlogN assigments shuttling the values between the two
217    * arrays. The number of compares appears to vary between N-1 and
218    * NlogN depending on the initial order but the main reason for
219    * using it here is that, unlike qsort, it is stable.
220    */
221   public void shuttlesort(int[] from, int[] to, int low, int high) {
222     if (high - low < 2) {
223       return;
224     }
225     int middle = (low + high) / 2;
226     shuttlesort(to, from, low, middle);
227     shuttlesort(to, from, middle, high);
228 
229     int p = low;
230     int q = middle;
231 
232     /*
233      * This is an optional short-cut; at each recursive call,
234      * check to see if the elements in this subset are already
235      * ordered.  If so, no further comparisons are needed; the
236      * sub-array can just be copied.  The array must be copied rather
237      * than assigned otherwise sister calls in the recursion might
238      * get out of sinc.  When the number of elements is three they
239      * are partitioned so that the first set, [low, mid), has one
240      * element and and the second, [mid, high), has two. We skip the
241      * optimisation when the number of elements is three or less as
242      * the first compare in the normal merge will produce the same
243      * sequence of steps. This optimisation seems to be worthwhile
244      * for partially ordered lists but some analysis is needed to
245      * find out how the performance drops to Nlog(N) as the initial
246      * order diminishes - it may drop very quickly.
247      */
248     if (high - low >= 4 && compare(from[middle - 1], from[middle]) <= 0) {
249       System.arraycopy(from, low, to, low, (high - low));
250       return;
251     }
252 
253     // A normal merge.
254 
255     for (int i = low; i < high; i++) {
256       if (q >= high || (p < middle && compare(from[p], from[q]) <= 0)) {
257         to[i] = from[p++];
258       } else {
259         to[i] = from[q++];
260       }
261     }
262   }

264   public void swap(int i, int j) {
265     int tmp = indexes[i];
266     indexes[i] = indexes[j];
267     indexes[j] = tmp;
268   }

301   public void sortByColumn(int column, boolean ascending) {
302     this.ascending = ascending;
303     sortingColumns.removeAllElements();

309   // There is no-where else to put this.
310   // Add a mouse listener to the Table to trigger a table sort
311   // when a column heading is clicked in the JTable.
312   public void addMouseListenerToHeaderInTable(JTable table) {
313     final TableSorter sorter = this;
314     final JTable tableView = table;
315     MouseAdapter listMouseListener = new MouseAdapter() {
316       @Override
317       public void mouseClicked(MouseEvent e) {
318         if (SwingUtilities.isLeftMouseButton(e)) {
319           try {
320             TableColumnModel columnModel = tableView.getColumnModel();
321             int viewColumn = columnModel.getColumnIndexAtX(e.getX());
322             int column = tableView.convertColumnIndexToModel(viewColumn);
323             if (e.getClickCount() == 1 && column != -1) {
324               int shiftPressed = e.getModifiers() & InputEvent.SHIFT_MASK;
325               boolean ascending = (shiftPressed == 0);
326               sorter.sortByColumn(column, ascending);
327             }
328           } catch (Exception ex) {
329             logger.error(ex.getMessage(), ex);
330           }
331         }
332       }
333     };
334     JTableHeader th = tableView.getTableHeader();
335     th.addMouseListener(listMouseListener);
336   }
