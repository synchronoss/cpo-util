21  package org.synchronoss.cpo.util;
22  
23  import org.synchronoss.cpo.meta.domain.CpoAttribute;
24  
25  import javax.swing.table.AbstractTableModel;
26  import java.util.*;
27  
28  public class CoreAttributeTableModel extends AbstractTableModel  {
29  
30    // Version Id for this class
31    private static final long serialVersionUID=1L;
32  
33    protected CpoAttributeLabelNode cpoAttLabNode;
34  
35    public CoreAttributeTableModel(CpoAttributeLabelNode cpoAttLabNode) {
36      this.cpoAttLabNode = cpoAttLabNode;
37    }
38  
39    public List<String> getColumnNames() {
40      return Arrays.asList("Attribute", "Data Name", "Data Type", "Transform Class", "Modified?");
41    }
42  
43    public List<Class<?>> getColumnClasses() {
44      Class<?>[] columnClasses = {String.class, String.class, String.class, String.class, String.class};
45      return Arrays.asList(columnClasses);
46    }
47  
48    public int getRowCount() {
49      return cpoAttLabNode.getChildCount();
50    }
51    
52    public int getColumnCount() {
53      return getColumnNames().size();
54    }
55  
56    @Override
57    public String getColumnName(int columnIndex) {
58      return getColumnNames().get(columnIndex);
59    }
60  
61    @Override
62    public Class<?> getColumnClass(int columnIndex) {
63      return getColumnClasses().get(columnIndex);
64    }
65  
66    @Override
67    public boolean isCellEditable(int rowIndex, int columnIndex) {
68      return (columnIndex < 4);
69    }
70  
71    @Override
72    public Object getValueAt(int rowIndex, int columnIndex) {
73      CpoAttributeNode cpoAttributeNode = (CpoAttributeNode)cpoAttLabNode.getChildAt(rowIndex);
74      CpoAttribute att = cpoAttributeNode.getUserObject();
75  
76      if (columnIndex == 0) {
77        return att.getJavaName();
78      } else if (columnIndex == 1) {
79        return att.getDataName();
80      } else if (columnIndex == 2) {
81        return att.getDataType();
82      } else if (columnIndex == 3) {
83        return att.getTransformClassName();
84      } else if (columnIndex == 4) {
85        if (cpoAttributeNode.isRemove()) {
86          return "Removed";
87        } else if (cpoAttributeNode.isNew()) {
88          return "New";
89        } else if (cpoAttributeNode.isDirty()) {
90          return "Changed";
91        } else {
92          return "";
93        }
94      } else {
95        return null;
96      }
97    }
98  
99    @Override
100   public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
101     String val = aValue.toString();
102     CpoAttributeNode cpoAttributeNode = (CpoAttributeNode)cpoAttLabNode.getChildAt(rowIndex);
103     if (columnIndex == 0) {
104       cpoAttributeNode.setAttribute(val);
105     } else if (columnIndex == 1) {
106       cpoAttributeNode.setDataName(val);
107     } else if (columnIndex == 2) {
108       cpoAttributeNode.setDataType(val);
109     } else if (columnIndex == 3) {
110       cpoAttributeNode.setTransformClass(val);
111     }
112     this.fireTableDataChanged();
113   }
114 }
