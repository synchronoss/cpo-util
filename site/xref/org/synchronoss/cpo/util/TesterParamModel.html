21  package org.synchronoss.cpo.util;
22  
23  import org.slf4j.*;
24  import org.synchronoss.cpo.meta.domain.CpoAttribute;
25  
26  import javax.swing.table.AbstractTableModel;
27  import java.util.*;
28  
29  public class TesterParamModel extends AbstractTableModel {
30  
31    // Version Id for this class
32    private static final long serialVersionUID = 1L;
33    private String[] columnNames = {"Function Seq Num", "Argument Seq Num", "Data Name", "Attribute Name", "Parameter"};
34    private Object[] columnClasses = {Integer.class, String.class, String.class, String.class, String.class};
35    private CpoFunctionGroupNode cpoFGnode;
36    private Hashtable<Object, Object> parameterMap = new Hashtable<Object, Object>();
37    private Logger logger = LoggerFactory.getLogger(this.getClass());
38  
39    public TesterParamModel(CpoFunctionGroupNode cpoQGnode) {
40      this.cpoFGnode = cpoQGnode;
41    }
42  
43    @Override
44    public int getRowCount() {
45      int rowCount = 0;
46      Enumeration<CpoFunctionNode> enumQueries = cpoFGnode.children();
47      while (enumQueries.hasMoreElements()) {
48        CpoFunctionNode node = enumQueries.nextElement();
49        rowCount = rowCount + node.getChildCount();
50      }
51      return rowCount;
52    }
53  
54    @Override
55    public int getColumnCount() {
56      return columnNames.length;
57    }
58  
59    @Override
60    public String getColumnName(int columnIndex) {
61      return columnNames[columnIndex];
62    }
63  
64    @Override
65    public Class<?> getColumnClass(int columnIndex) {
66      return (Class<?>)columnClasses[columnIndex];
67    }
68  
69    @Override
70    public boolean isCellEditable(int rowIndex, int columnIndex) {
71      return (columnIndex == 4);
72    }
73  
74    @Override
75    public Object getValueAt(int rowIndex, int columnIndex) {
76      int rowCount = 0;
77      Enumeration<CpoFunctionNode> enumQueries = cpoFGnode.children();
78      while (enumQueries.hasMoreElements()) {
79        CpoFunctionNode node = enumQueries.nextElement();
80        Enumeration<CpoArgumentNode> enumArguments = node.children();
81        while (enumArguments.hasMoreElements()) {
82          CpoArgumentNode argumentNode = enumArguments.nextElement();
83          if (rowCount == rowIndex) {
84            if (columnIndex == 0) {
85              return node.getSeqNo();
86            } else if (columnIndex == 1) {
87              return argumentNode.getSeqNo();
88            } else if (columnIndex == 2) {
89              return argumentNode.getUserObject().getAttribute().getDataName();
90            } else if (columnIndex == 3) {
91              return argumentNode.getUserObject().getAttribute().getJavaName();
92            } else if (columnIndex == 4) {
93              CpoAttribute attribute = argumentNode.getUserObject().getAttribute();
94              return this.parameterMap.get(attribute != null ? attribute.getJavaName() : "");
95            }
96          }
97          rowCount++;
98        }
99      }
100     return null;
101   }
102 
103   @Override
104   public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
105     if (columnIndex == 4) {
106       logger.debug(getValueAt(rowIndex, 3) + " : " + aValue);
107       if (aValue == null) {
108         this.parameterMap.remove(getValueAt(rowIndex, 3));
109       }
110 
111       this.parameterMap.put(getValueAt(rowIndex, 3), aValue);
112       this.fireTableDataChanged();
113     }
114   }
115 
116   protected boolean isTableFilledOut() {
117     int rows = getRowCount();
118     for (int i = 0; i < rows; i++) {
119       if (getValueAt(i, 4) == null || getValueAt(i, 4).equals("")) {
120         return false;
121       }
122     }
123     return true;
124   }
125 
126   public Hashtable<Object, Object> getParameterMap() {
127     return parameterMap;
128   }
129 }
