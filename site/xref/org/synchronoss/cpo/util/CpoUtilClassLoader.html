21  package org.synchronoss.cpo.util;
22  
23  import org.slf4j.*;
24  
25  import java.io.*;
26  import java.util.*;
27  import java.util.jar.*;
28  import java.util.zip.ZipEntry;
29  
30  public class CpoUtilClassLoader extends ClassLoader {
31  
32    private static CpoUtilClassLoader loader;
33    private static final Object LOCK = new Object();
34    private Logger logger = LoggerFactory.getLogger(this.getClass());
35  
36    private CpoUtilClassLoader(ClassLoader parent) {
37      super(parent);
38    }
39  
40    public static CpoUtilClassLoader getInstance(ClassLoader parent) {
41      synchronized (LOCK) {
42        if (loader == null) {
43          loader = new CpoUtilClassLoader(parent);
44        }
45      }
46      return loader;
47    }
48  
49    public static void unloadLoader() {
50      synchronized (LOCK) {
51        loader = null;
52      }
53    }
54  
55    @Override
56    protected Class<?> findClass(String name) throws ClassNotFoundException {
57      Class<?> c;
58      // Convert class name argument to filename
59      // Convert package names into subdirectories
60      try {
61        byte[] data = loadClassData(name);
62        c = defineClass(name, data, 0, data.length);
63        if (c == null) {
64          throw new ClassNotFoundException(name);
65        }
66        if (logger.isDebugEnabled()) {
67          logger.debug("loaded class: " + name);
68        }
69      } catch (IOException e) {
70        logger.error(e.getMessage(), e);
71        throw new ClassNotFoundException("Error reading class: " + name, e);
72      }
73      return c;
74    }
75  
76    private byte[] loadClassData(String name) throws IOException {
77      for (File file : CpoUtil.getInstance().getCustomClasspathEntries()) {
78        if (file.isFile()) {
79          if (file.getName().toLowerCase().endsWith(".jar")) {
80            String filename = name.replace('.', '/') + ".class";
81            JarFile jf = new JarFile(file);
82            Enumeration<JarEntry> e = jf.entries();
83            while (e.hasMoreElements()) {
84              ZipEntry entry = e.nextElement();
85              if (filename.equals(entry.getName())) {
86                InputStream is = jf.getInputStream(entry);
87                int l = (int)entry.getSize();
88                byte[] buff = new byte[l];
89                int read = 0;
90                while (read < l) {
91                  int incr = is.read(buff, read, l - read);
92                  read += incr;
93                }
94                return buff;
95              }
96            }
97          }
98        } else if (file.isDirectory()) {
99          String filename = name.replace('.', File.separatorChar) + ".class";
100         File f = new File(file, filename);
101 
102         // Get size of class file
103         int size = (int)f.length();
104 
105         // Reserve space to read
106         byte[] buff = new byte[size];
107 
108         // Get stream to read from
109         FileInputStream fis = new FileInputStream(f);
110         DataInputStream dis = new DataInputStream(fis);
111 
112         // Read in data
113         dis.readFully(buff);
114 
115         // close stream
116         dis.close();
117 
118         // return data
119         return buff;
120       }
121     }
122     throw new IOException("Could not find class");
123   }
